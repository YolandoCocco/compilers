  - Criar uma gramática

prog: includes? globals? functions? main
includes: (INCLUDE STR)+
globals: GLOB OP vars CL
vars : (type ids EOL)+
ids : ID | ID SEP ids
functions : function+
function : 
   FUNCTION ID OPP ((type ID SEP)* type ID)? CLP block
main : MAIN block
block: OPCB line+ CLCB
line : decl | input | output | atrib | uif | uwhile | udo   


INCLUDE: '#include'
STR: '"'.?*'"'
GLOB: 'global'
OP : '{'
CL : '}'
SEP: ','
